C_Command = LineRange Command
	  |           Command
	  | LineRange

LineRange = Line
	  | [Line] ',' [Line]

     Line = Digits
          | Symbol

   Symbol = '$' | '.'

   Digits = 0-9 {0-9}

  Command = 'p' [PolyArg] 
	  | 's' [PolyArg]
	  | 'g' [PolyArg]
	  | 'w' [MonoArg] 
	  | 'r' [MonoArg] 
	  | 'e' [MonoArg] 
          | 'd' 
	  | 'q' 
	  | ''

  PolyArg = Arg {Arg}

  MonoArg = Arg
